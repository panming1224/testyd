    print("✅ 有库存，可以购买")

### 第4步：学会循环处理（5分钟）

创建文件 `D:/testyd/教学副本/练习/loops.py`：

```python
# 循环 - 重复执行任务

# 处理商品列表
products = ["手机", "电脑", "平板", "耳机"]

print("📋 商品清单：")
for product in products:
    print(f"  - {product}")

# 带编号的循环
print("\n📋 带编号的商品清单：")
for i, product in enumerate(products, 1):
    print(f"  {i}. {product}")

# 处理字典数据
shop_data = {
    "shop1": {"name": "数码专营店", "sales": 1000},
    "shop2": {"name": "家电旗舰店", "sales": 2000},
    "shop3": {"name": "服装精品店", "sales": 1500}
}

print("\n📊 店铺销售数据：")
for shop_id, info in shop_data.items():
    name = info["name"]
    sales = info["sales"]
    print(f"  {shop_id}: {name} - 销售额：{sales}元")

# 数字循环
print("\n⏰ 倒计时：")
for i in range(5, 0, -1):
    print(f"  {i}...")
print("  🚀 开始下载！")
```

### 常见错误和解决方法

#### 1. 语法错误
```python
# ❌ 错误：缺少冒号
if age > 18
    print("成年人")

# ✅ 正确：
if age > 18:
    print("成年人")
```

#### 2. 缩进错误
```python
# ❌ 错误：缩进不一致
if True:
    print("第一行")
  print("第二行")  # 缩进错误

# ✅ 正确：
if True:
    print("第一行")
    print("第二行")
```

#### 3. 变量名错误
```python
# ❌ 错误：变量名拼写错误
shop_name = "测试店铺"
print(shop_nam)  # 拼写错误

# ✅ 正确：
shop_name = "测试店铺"
print(shop_name)
```

### 代码规范建议
```python
# 1. 使用有意义的变量名
# ❌ 不好的命名
a = "店铺1"
b = 100

# ✅ 好的命名
shop_name = "店铺1"
product_count = 100

# 2. 添加注释
def calculate_total(price, quantity):
    """
    计算总价
    参数：
        price: 单价
        quantity: 数量
    返回：
        总价
    """
    return price * quantity

# 3. 保持代码简洁
# ❌ 复杂的写法
if status == "success":
    result = True
else:
    result = False

# ✅ 简洁的写法
result = (status == "success")
```
